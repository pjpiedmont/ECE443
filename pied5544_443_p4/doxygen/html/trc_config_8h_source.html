67  * Configuration Macro: TRC_CFG_HARDWARE_PORT

69  * Specify what hardware port to use (i.e., the "timestamping driver").
70  *
71  * All ARM Cortex-M MCUs are supported by "TRC_HARDWARE_PORT_ARM_Cortex_M".

141  /******************************************************************************
142  * TRC_CFG_INCLUDE_MEMMANG_EVENTS
143  *
144  * Macro which should be defined as either zero (0) or one (1).
145  *
146  * This controls if malloc and free calls should be traced. Set this to zero (0)
147  * to exclude malloc/free calls, or one (1) to include such events in the trace.
148  *
149  * Default value is 1.
150  *****************************************************************************/
151 #define TRC_CFG_INCLUDE_MEMMANG_EVENTS 1
152 
153  /******************************************************************************
154  * TRC_CFG_INCLUDE_USER_EVENTS
155  *
156  * Macro which should be defined as either zero (0) or one (1).
157  *
158  * If this is zero (0), all code related to User Events is excluded in order
159  * to reduce code size. Any attempts of storing User Events are then silently
160  * ignored.
161  *
162  * User Events are application-generated events, like "printf" but for the
163  * trace log, generated using vTracePrint and vTracePrintF.
164  * The formatting is done on host-side, by Tracealyzer. User Events are
165  * therefore much faster than a console printf and can often be used
166  * in timing critical code without problems.
167  *
168  * Note: In streaming mode, User Events are used to provide error messages
169  * and warnings from the recorder (in case of incorrect configuration) for
170  * display in Tracealyzer. Disabling user events will also disable these
171  * warnings. You can however still catch them by calling xTraceGetLastError
172  * or by putting breakpoints in prvTraceError and prvTraceWarning.
173  *
174  * Default value is 1.
175  *****************************************************************************/
176 #define TRC_CFG_INCLUDE_USER_EVENTS 1
177 
178  /*****************************************************************************
179  * TRC_CFG_INCLUDE_ISR_TRACING
180  *
181  * Macro which should be defined as either zero (0) or one (1).
182  *
183  * If this is zero (0), the code for recording Interrupt Service Routines is
184  * excluded, in order to reduce code size. This means that any calls to
185  * vTraceStoreISRBegin/vTraceStoreISREnd will be ignored.
186  * This does not completely disable ISR tracing, in cases where an ISR is
187  * calling a traced kernel service. These events will still be recorded and
188  * show up in anonymous ISR instances in Tracealyzer, with names such as
189  * "ISR sending to <queue name>".
190  * To disable such tracing, please refer to vTraceSetFilterGroup and
191  * vTraceSetFilterMask.
192  *
193  * Default value is 1.
194  *
195  * Note: tracing ISRs requires that you insert calls to vTraceStoreISRBegin
196  * and vTraceStoreISREnd in your interrupt handlers.
197  *****************************************************************************/
198 #define TRC_CFG_INCLUDE_ISR_TRACING 1
199 
200  /*****************************************************************************
201  * TRC_CFG_INCLUDE_READY_EVENTS
202  *
203  * Macro which should be defined as either zero (0) or one (1).
204  *
205  * If one (1), events are recorded when tasks enter scheduling state "ready".
206  * This allows Tracealyzer to show the initial pending time before tasks enter
207  * the execution state, and present accurate response times.
208  * If zero (0), "ready events" are not created, which allows for recording
209  * longer traces in the same amount of RAM.
210  *
211  * Default value is 1.
212  *****************************************************************************/
213 #define TRC_CFG_INCLUDE_READY_EVENTS 1
214 
215  /*****************************************************************************
216  * TRC_CFG_INCLUDE_OSTICK_EVENTS
217  *
218  * Macro which should be defined as either zero (0) or one (1).
219  *
220  * If this is one (1), events will be generated whenever the OS clock is
221  * increased. If zero (0), OS tick events are not generated, which allows for
222  * recording longer traces in the same amount of RAM.
223  *
224  * Default value is 1.
225  *****************************************************************************/
226 #define TRC_CFG_INCLUDE_OSTICK_EVENTS 1

274  /******************************************************************************
275  * TRC_CFG_ENABLE_STACK_MONITOR
276  *
277  * If enabled (1), the recorder periodically reports the unused stack space of
278  * all active tasks.
279  * The stack monitoring runs in the Tracealyzer Control task, TzCtrl. This task
280  * is always created by the recorder when in streaming mode.
281  * In snapshot mode, the TzCtrl task is only used for stack monitoring and is
282  * not created unless this is enabled.
283  *****************************************************************************/
284 #define TRC_CFG_ENABLE_STACK_MONITOR 1
285 
286  /******************************************************************************
287  * TRC_CFG_STACK_MONITOR_MAX_TASKS
288  *
289  * Macro which should be defined as a non-zero integer value.
290  *
291  * This controls how many tasks that can be monitored by the stack monitor.
292  * If this is too small, some tasks will be excluded and a warning is shown.
293  *
294  * Default value is 10.
295  *****************************************************************************/
296 #define TRC_CFG_STACK_MONITOR_MAX_TASKS 10
297 
298  /******************************************************************************
299  * TRC_CFG_STACK_MONITOR_MAX_REPORTS
300  *
301  * Macro which should be defined as a non-zero integer value.
302  *
303  * This defines how many tasks that will be subject to stack usage analysis for
304  * each execution of the Tracealyzer Control task (TzCtrl). Note that the stack
305  * monitoring cycles between the tasks, so this does not affect WHICH tasks that
306  * are monitored, but HOW OFTEN each task stack is analyzed.
307  *
308  * This setting can be combined with TRC_CFG_CTRL_TASK_DELAY to tune the
309  * frequency of the stack monitoring. This is motivated since the stack analysis
310  * can take some time to execute.
311  * However, note that the stack analysis runs in a separate task (TzCtrl) that
312  * can be executed on low priority. This way, you can avoid that the stack
313  * analysis disturbs any time-sensitive tasks.
314  *
315  * Default value is 1.
316  *****************************************************************************/
317 #define TRC_CFG_STACK_MONITOR_MAX_REPORTS 1
318 
319  /*******************************************************************************
320  * Configuration Macro: TRC_CFG_CTRL_TASK_PRIORITY
321  *
322  * The scheduling priority of the Tracealyzer Control (TzCtrl) task.
323  *
324  * In streaming mode, TzCtrl is used to receive start/stop commands from
325  * Tracealyzer and in some cases also to transmit the trace data (for stream
326  * ports that uses the internal buffer, like TCP/IP). For such stream ports,
327  * make sure the TzCtrl priority is high enough to ensure reliable periodic
328  * execution and transfer of the data, but low enough to avoid disturbing any
329  * time-sensitive functions.
330  *
331  * In Snapshot mode, TzCtrl is only used for the stack usage monitoring and is
332  * not created if stack monitoring is disabled. TRC_CFG_CTRL_TASK_PRIORITY should
333  * be low, to avoid disturbing any time-sensitive tasks.
334  ******************************************************************************/
335 #define TRC_CFG_CTRL_TASK_PRIORITY 1
336 
337  /*******************************************************************************
338  * Configuration Macro: TRC_CFG_CTRL_TASK_DELAY
339  *
340  * The delay between loops of the TzCtrl task (see TRC_CFG_CTRL_TASK_PRIORITY),
341  * which affects the frequency of the stack monitoring.
342  *
343  * In streaming mode, this also affects the trace data transfer if you are using
344  * a stream port leveraging the internal buffer (like TCP/IP). A shorter delay
345  * increases the CPU load of TzCtrl somewhat, but may improve the performance of
346  * of the trace streaming, especially if the trace buffer is small.
347  ******************************************************************************/
348 #define TRC_CFG_CTRL_TASK_DELAY 10
349 
350  /*******************************************************************************
351  * Configuration Macro: TRC_CFG_CTRL_TASK_STACK_SIZE
352  *
353  * The stack size of the Tracealyzer Control (TzCtrl) task.
354  * See TRC_CFG_CTRL_TASK_PRIORITY for further information about TzCtrl.
355  ******************************************************************************/
356 #define TRC_CFG_CTRL_TASK_STACK_SIZE (configMINIMAL_STACK_SIZE * 2)
357 
358 /*******************************************************************************
359  * Configuration Macro: TRC_CFG_RECORDER_BUFFER_ALLOCATION
360  *
361  * Specifies how the recorder buffer is allocated (also in case of streaming, in
362  * port using the recorder's internal temporary buffer)
363  *
364  * Values:
365  * TRC_RECORDER_BUFFER_ALLOCATION_STATIC - Static allocation (internal)
366  * TRC_RECORDER_BUFFER_ALLOCATION_DYNAMIC - Malloc in vTraceEnable
367  * TRC_RECORDER_BUFFER_ALLOCATION_CUSTOM - Use vTraceSetRecorderDataBuffer
368  *
369  * Static and dynamic mode does the allocation for you, either in compile time
370  * (static) or in runtime (malloc).
371  * The custom mode allows you to control how and where the allocation is made,
372  * for details see TRC_ALLOC_CUSTOM_BUFFER and vTraceSetRecorderDataBuffer().
373  ******************************************************************************/
374 #define TRC_CFG_RECORDER_BUFFER_ALLOCATION TRC_RECORDER_BUFFER_ALLOCATION_STATIC
375 
376 /******************************************************************************
377  * TRC_CFG_MAX_ISR_NESTING
378  *
379  * Defines how many levels of interrupt nesting the recorder can handle, in
380  * case multiple ISRs are traced and ISR nesting is possible. If this
381  * is exceeded, the particular ISR will not be traced and the recorder then

383  * for keeping track of the previous execution context (4 byte per entry).
384  *
385  * This value must be a non-zero positive constant, at least 1.
386  *
387  * Default value: 8
388  *****************************************************************************/
389 #define TRC_CFG_MAX_ISR_NESTING 8
