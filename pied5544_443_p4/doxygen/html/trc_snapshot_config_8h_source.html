52 /******************************************************************************
53  * TRC_CFG_SNAPSHOT_MODE
54  *
55  * Macro which should be defined as one of:
56  * - TRC_SNAPSHOT_MODE_RING_BUFFER
57  * - TRC_SNAPSHOT_MODE_STOP_WHEN_FULL
58  * Default is TRC_SNAPSHOT_MODE_RING_BUFFER.
59  *
60  * With TRC_CFG_SNAPSHOT_MODE set to TRC_SNAPSHOT_MODE_RING_BUFFER, the
61  * events are stored in a ring buffer, i.e., where the oldest events are
62  * overwritten when the buffer becomes full. This allows you to get the last
63  * events leading up to an interesting state, e.g., an error, without having
64  * to store the whole run since startup.
65  *
66  * When TRC_CFG_SNAPSHOT_MODE is TRC_SNAPSHOT_MODE_STOP_WHEN_FULL, the
67  * recording is stopped when the buffer becomes full. This is useful for
68  * recording events following a specific state, e.g., the startup sequence.
69  *****************************************************************************/
70 #define TRC_CFG_SNAPSHOT_MODE TRC_SNAPSHOT_MODE_RING_BUFFER
71 
72 /*******************************************************************************
73  * TRC_CFG_EVENT_BUFFER_SIZE
74  *
75  * Macro which should be defined as an integer value.
76  *
77  * This defines the capacity of the event buffer, i.e., the number of records
78  * it may store. Most events use one record (4 byte), although some events
79  * require multiple 4-byte records. You should adjust this to the amount of RAM
80  * available in the target system.
81  *
82  * Default value is 1000, which means that 4000 bytes is allocated for the
83  * event buffer.
84  ******************************************************************************/
85 #define TRC_CFG_EVENT_BUFFER_SIZE 1000

87 /*******************************************************************************
88  * TRC_CFG_NTASK, TRC_CFG_NISR, TRC_CFG_NQUEUE, TRC_CFG_NSEMAPHORE...
89  *
90  * A group of macros which should be defined as integer values, zero or larger.
91  *
92  * These define the capacity of the Object Property Table, i.e., the maximum
93  * number of objects active at any given point, within each object class (e.g.,
94  * task, queue, semaphore, ...).
95  *
96  * If tasks or other objects are deleted in your system, this
97  * setting does not limit the total amount of objects created, only the number
98  * of objects that have been successfully created but not yet deleted.
99  *
100  * Using too small values will cause vTraceError to be called, which stores an
101  * error message in the trace that is shown when opening the trace file. The
102  * error message can also be retrieved using xTraceGetLastError.
103  *
104  * It can be wise to start with large values for these constants,
105  * unless you are very confident on these numbers. Then do a recording and
106  * check the actual usage by selecting View menu -> Trace Details ->
107  * Resource Usage -> Object Table.
108  ******************************************************************************/

194 /******************************************************************************
195  * TRC_CFG_USE_IMPLICIT_IFE_RULES
196  *
197  * Macro which should be defined as either zero (0) or one (1).
198  * Default is 1.
199  *
200  * Tracealyzer groups the events into "instances" based on Instance Finish
201  * Events (IFEs), produced either by default rules or calls to the recorder
202  * functions vTraceInstanceFinishedNow and vTraceInstanceFinishedNext.
203  *
204  * If TRC_CFG_USE_IMPLICIT_IFE_RULES is one (1), the default IFE rules is
205  * used, resulting in a "typical" grouping of events into instances.
206  * If these rules don't give appropriate instances in your case, you can
207  * override the default rules using vTraceInstanceFinishedNow/Next for one
208  * or several tasks. The default IFE rules are then disabled for those tasks.
209  *
210  * If TRC_CFG_USE_IMPLICIT_IFE_RULES is zero (0), the implicit IFE rules are
211  * disabled globally. You must then call vTraceInstanceFinishedNow or
212  * vTraceInstanceFinishedNext to manually group the events into instances,
213  * otherwise the tasks will appear a single long instance.
214  *
215  * The default IFE rules count the following events as "instance finished":
216  * - Task delay, delay until
217  * - Task suspend
218  * - Blocking on "input" operations, i.e., when the task is waiting for the
219  * next a message/signal/event. But only if this event is blocking.
220  *
221  * For details, see trcSnapshotKernelPort.h and look for references to the
222  * macro trcKERNEL_HOOKS_SET_TASK_INSTANCE_FINISHED.
223  *****************************************************************************/
224 #define TRC_CFG_USE_IMPLICIT_IFE_RULES 1
225 
226 /******************************************************************************
227  * TRC_CFG_USE_16BIT_OBJECT_HANDLES
228  *
229  * Macro which should be defined as either zero (0) or one (1).
230  *
231  * If set to 0 (zero), the recorder uses 8-bit handles to identify kernel
232  * objects such as tasks and queues. This limits the supported number of
233  * concurrently active objects to 255 of each type (tasks, queues, mutexes,
234  * etc.) Note: 255, not 256, since handle 0 is reserved.
235  *
236  * If set to 1 (one), the recorder uses 16-bit handles to identify kernel
237  * objects such as tasks and queues. This limits the supported number of
238  * concurrent objects to 65535 of each type (object class). However, since the
239  * object property table is limited to 64 KB, the practical limit is about
240  * 3000 objects in total.
241  *
242  * Default is 0 (8-bit handles)
243  *
244  * NOTE: An object with handle above 255 will use an extra 4-byte record in
245  * the event buffer whenever the object is referenced. Moreover, some internal
246  * tables in the recorder gets slightly larger when using 16-bit handles.
247  *****************************************************************************/
248 #define TRC_CFG_USE_16BIT_OBJECT_HANDLES 0

256  * If this is one (1), the TRACE_ASSERT macro (used at various locations in the

268 /*******************************************************************************
269  * TRC_CFG_USE_SEPARATE_USER_EVENT_BUFFER
270  *
271  * Macro which should be defined as an integer value.
272  *
273  * Set TRC_CFG_USE_SEPARATE_USER_EVENT_BUFFER to 1 to enable the
274  * separate user event buffer (UB).
275  * In this mode, user events are stored separately from other events,
276  * e.g., RTOS events. Thereby you can get a much longer history of
277  * user events as they don't need to share the buffer space with more
278  * frequent events.
279  *
280  * The UB is typically used with the snapshot ring-buffer mode, so the
281  * recording can continue when the main buffer gets full. And since the
282  * main buffer then overwrites the earliest events, Tracealyzer displays
283  * "Unknown Actor" instead of task scheduling for periods with UB data only.
284  *
285  * In UB mode, user events are structured as UB channels, which contains
286  * a channel name and a default format string. Register a UB channel using
287  * xTraceRegisterUBChannel.
288  *
289  * Events and data arguments are written using vTraceUBEvent and
290  * vTraceUBData. They are designed to provide efficient logging of
291  * repeating events, using the same format string within each channel.
292  *
293  * Examples:
294  *
295  * traceString chn1 = xTraceRegisterString("Channel 1");
296  * traceString fmt1 = xTraceRegisterString("Event!");
297  * traceUBChannel UBCh1 = xTraceRegisterUBChannel(chn1, fmt1);
298  *
299  * traceString chn2 = xTraceRegisterString("Channel 2");
300  * traceString fmt2 = xTraceRegisterString("X: %d, Y: %d");
301  * traceUBChannel UBCh2 = xTraceRegisterUBChannel(chn2, fmt2);
302  *
303  * // Result in "[Channel 1] Event!"
304  * vTraceUBEvent(UBCh1);
305  *
306  * // Result in "[Channel 2] X: 23, Y: 19"
307  * vTraceUBData(UBCh2, 23, 19);
308  *
309  * You can also use the other user event functions, like vTracePrintF.
310  * as they are then rerouted to the UB instead of the main event buffer.
311  * vTracePrintF then looks up the correct UB channel based on the
312  * provided channel name and format string, or creates a new UB channel
313  * if no match is found. The format string should therefore not contain
314  * "random" messages but mainly format specifiers. Random strings should
315  * be stored using %s and with the string as an argument.
316  *
317  * // Creates a new UB channel ("Channel 2", "%Z: %d")
318  * vTracePrintF(chn2, "%Z: %d", value1);
319  *
320  * // Finds the existing UB channel
321  * vTracePrintF(chn2, "%Z: %d", value2);
322 
323  ******************************************************************************/
324 #define TRC_CFG_USE_SEPARATE_USER_EVENT_BUFFER 0

352 /*******************************************************************************
353  * TRC_CFG_ISR_TAILCHAINING_THRESHOLD
354  *
355  * Macro which should be defined as an integer value.
356  *
357  * If tracing multiple ISRs, this setting allows for accurate display of the
358  * context-switching also in cases when the ISRs execute in direct sequence.
359  *
360  * vTraceStoreISREnd normally assumes that the ISR returns to the previous
361  * context, i.e., a task or a preempted ISR. But if another traced ISR
362  * executes in direct sequence, Tracealyzer may incorrectly display a minimal
363  * fragment of the previous context in between the ISRs.
364  *
365  * By using TRC_CFG_ISR_TAILCHAINING_THRESHOLD you can avoid this. This is
366  * however a threshold value that must be measured for your specific setup.
367  * See http://percepio.com/2014/03/21/isr_tailchaining_threshold/
368  *
369  * The default setting is 0, meaning "disabled" and that you may get an
370  * extra fragments of the previous context in between tail-chained ISRs.
371  *
372  * Note: This setting has separate definitions in trcSnapshotConfig.h and
373  * trcStreamingConfig.h, since it is affected by the recorder mode.
374  ******************************************************************************/
375 #define TRC_CFG_ISR_TAILCHAINING_THRESHOLD 0
